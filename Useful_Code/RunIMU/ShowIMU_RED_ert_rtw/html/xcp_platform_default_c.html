20/* Define the API to pause the execution (sleep) for a specific amount of time */



21#if defined(_WIN32) || defined(__WIN32__) || defined(WIN32)



22



23#define XCP_MICROSECONDS_PER_SECOND  1000000



24



25static boolean_T xcpPerfCounterAvailable      = true;



26static boolean_T xcpPerfCounterInitialized    = false;



27static uint64_t  xcpPerfCounterTicksPerSecond = 1;



28



29/* Initialize performance counters */



30static void xcpPerfCounterInit(void)



31{



32    if (!xcpPerfCounterInitialized) {



33        LARGE_INTEGER frequency;



34



35        if (QueryPerformanceFrequency(&frequency)) {



36            if ((frequency.u.HighPart != 0) || (frequency.u.LowPart == 0)) {



37                XCP_PRINTF("xcpTimestampInit: QueryPerformanceFrequency error, unexpected frequency\n");



38                xcpPerfCounterAvailable = false;



39            } else {



40                xcpPerfCounterTicksPerSecond = frequency.u.LowPart;



41            }



42        }



43        else {



44            XCP_PRINTF("xcpTimestampInit: QueryPerformanceFrequency error, performance Counter not available\n");



45            xcpPerfCounterAvailable = false;



46        }



47



48        xcpPerfCounterInitialized = true;



49    }



50}



51



52



53/* Convert a performance counter value (in ticks) into a value according to a specific time unit */



54static uint64_t xcpPerfCounterValue2TimeUnits(LARGE_INTEGER value, uint64_t timeUnitsPerSecond)



55{



56    return ((((uint64_t)value.u.HighPart) << 32) | (uint64_t)value.u.LowPart) * timeUnitsPerSecond / xcpPerfCounterTicksPerSecond;



57}

61void xcpSleep(uint32_T seconds, uint32_T microseconds)



62{



63    if ((seconds == 0) && (microseconds == 0)) {



64        /* According to MSDN, a value of zero causes the thread to relinquish



65           the remainder of its time slice to any other thread that is ready to run.



66           If there are no other threads ready to run, the function returns immediately



67           and the thread continues execution */



68        Sleep(0);



69    } else if ((seconds == 0) && (microseconds < 1000)) {



70        /* Sleep() doesn't allow to sleep for less than 1ms.



71           If a sleep time of less than 1ms is requested on Windows, a busy wait



72           based on performance counters is used instead */



73        if (!xcpPerfCounterInitialized) {



74            /* Initialize performance counters */



75            xcpPerfCounterInit();



76        }



77



78        if (xcpPerfCounterAvailable) {



79            LARGE_INTEGER current;



80            uint64_t startTime = 0;



81            uint64_t currentTime = 0;



82            uint64_t elapsedTime = 0;



83



84            QueryPerformanceCounter(&current);



85            startTime = xcpPerfCounterValue2TimeUnits(current, XCP_MICROSECONDS_PER_SECOND);



86



87            do



88            {



89                QueryPerformanceCounter(&current);



90                currentTime = xcpPerfCounterValue2TimeUnits(current, XCP_MICROSECONDS_PER_SECOND);



91



92                if (currentTime >= startTime) {



93                    elapsedTime = currentTime - startTime;



94                } else {



95                    elapsedTime = currentTime + UINT64_MAX - startTime;



96                }



97            }



98            while(elapsedTime <= microseconds);



99        }



100    } else {



101        /* Sleep for the required amount of ms */



102        Sleep((seconds) * 1000 + (((microseconds) + 1000 - 1) / 1000));



103    }



104}

108void xcpSleep(uint32_T seconds, uint32_T microseconds)



109{



110#if _POSIX_C_SOURCE >= 199309L



111	/* nanosleep() is the preferred sleep method on Linux */



112	struct timespec t;



113	t.tv_sec = seconds;



114	t.tv_nsec = microseconds * 1000;



115



116	nanosleep(&t, NULL);



117#else



118	/* if nanosleep() is not available, use the select() instead */



119	struct timeval t;



120	t.tv_sec = seconds;



121	t.tv_usec = (suseconds_t)microseconds; /* tv_usec is signed */



122



123	select(0, NULL, NULL, NULL, &t);



124#endif /* _POSIX_C_SOURCE >= 199309L */



125}

183/* Initialize timestamp based on performance counters */



184static void xcpTimestampInit(void)



185{



186    if (!xcpTimestampInitialized) {



187        if (!xcpPerfCounterInitialized) {



188            /* Initialize performance counters */



189            xcpPerfCounterInit();



190        }



191



192        if (xcpPerfCounterAvailable) {



193            LARGE_INTEGER current;



194



195            /* Initialize xcpTimestampStartTime based on current time */



196            QueryPerformanceCounter(&current);



197



198            xcpTimestampStartTime = xcpPerfCounterValue2TimeUnits(current, XCP_TIMESTAMP_UNITS_PER_SECOND);



199        }



200



201        xcpTimestampInitialized = true;



202    }



203}



204



205/* Get timestamp value in XCP_TIMESTAMP_UNITs



206    @note in the current implementation the time starts counting from 0 on the first call */



207uint32_T xcpGetTimestamp(void)



208{



209    LARGE_INTEGER current;



210    uint64_t currentTime;



211    uint32_T ret = 0;



212



213    if (!xcpTimestampInitialized) {



214        xcpTimestampInit();



215        return ret;



216    }



217



218    if (xcpPerfCounterAvailable) {



219        QueryPerformanceCounter(&current);



220        currentTime = xcpPerfCounterValue2TimeUnits(current, XCP_TIMESTAMP_UNITS_PER_SECOND);



221



222        if (currentTime >= xcpTimestampStartTime) {



223            ret = (uint32_T) (currentTime - xcpTimestampStartTime);



224        } else {



225            ret = (uint32_T) (currentTime + UINT64_MAX - xcpTimestampStartTime);



226        }



227    }



228



229    return ret;



230}

234/* Return value in XCP_TIMESTAMP_UNITs



235   @note in the current implementation the time starts counting from 0 on the first call */



236uint32_T xcpGetTimestamp(void)



237{



238    struct timeval timeVal;



239    uint64_t currentTime = 0;



240    uint32_T ret = 0;



241



242    gettimeofday(&timeVal, NULL);



243    currentTime = timeVal.tv_sec * XCP_TIMESTAMP_UNITS_PER_SECOND + timeVal.tv_usec * XCP_TIMESTAMP_UNITS_PER_MICROSECOND;



244



245    if (!xcpTimestampInitialized) {



246        /* Read initial time and store it in xcpTimestampStartTime variable */



247        xcpTimestampStartTime = currentTime;



248



249        xcpTimestampInitialized = true;



250    }



251



252    if (currentTime >= xcpTimestampStartTime) {



253        ret = (uint32_T) (currentTime - xcpTimestampStartTime);



254    } else {



255        ret = (uint32_T) (currentTime + UINT64_MAX - xcpTimestampStartTime);



256    }



257



258    return ret;



259}

349#define MAX_EXECUTABLE_PATH_SIZE 1024



350#define UNINITIALIZED_ADDRESS ((intptr_t) -1)



351



352/* Get the dynamic image slide of system when the executable is deployed */



353uintptr_t xcpExecutableStartAddress(void)



354{    



355    static intptr_t imgSlideValue = UNINITIALIZED_ADDRESS;



356    



357    if (imgSlideValue == UNINITIALIZED_ADDRESS) {



358        char path[MAX_EXECUTABLE_PATH_SIZE];



359        uint32_t size = sizeof(path);



360        path[MAX_EXECUTABLE_PATH_SIZE-1] = '\0';



361        /* _NSGetExecutablePath returns 0 if the path of the main executable



362         * is successfully copied */



363        if (_NSGetExecutablePath(path, &size) == 0) {



364            for (uint32_t i = 0; (i < _dyld_image_count()) && (imgSlideValue == UNINITIALIZED_ADDRESS); i++)



365            {



366             /* If the path of the executable is the same as the indexed image



367             that means the image is the ASLR slide */



368                if (strcmp(_dyld_get_image_name(i), path) == 0)



369                    imgSlideValue = _dyld_get_image_vmaddr_slide(i);



370            }



371        }



372    }



373    return (uintptr_t)imgSlideValue;



374}

378    return (uint8_T*) ( ( (uintptr_t)address | ( ((uintptr_t)addressExtension) << 32) ) + xcpExecutableStartAddress() );
